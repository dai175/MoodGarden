> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Transform MoodGarden from a functional MVP into a living, quiet landscape by fixing placement, animations, and interaction design.

**Architecture:** Three sequential workstreams: (1) Wind animation system replacing unnatural pulse/scale animations across all 14 elements, (2) Placement tightening and visual cohesion, (3) UI interaction redesign (selector + post-recording transition). **Tech Stack:** SpriteKit (SKAction, SKScene.update), SwiftUI (gesture handling), Swift Testing framework

```swift
// MoodGarden/Garden/WindState.swift
import CoreGraphics
import Foundation

struct WindState {
    static let minStrength: CGFloat = 0.05
    static let maxStrength: CGFloat = 0.3

    private(set) var direction: CGFloat = 0.0
    private(set) var strength: CGFloat = minStrength

    private let strengthPeriod: TimeInterval = 10.0
    private let directionPeriod: TimeInterval = 25.0
    private var firstTime: TimeInterval = 0
    private var initialized = false

    mutating func update(currentTime: TimeInterval) {
        if !initialized {
            firstTime = currentTime
            initialized = true
            return
        }

        let elapsed = currentTime - firstTime

        // Sine-based strength oscillation
        let strengthPhase = elapsed / strengthPeriod * 2.0 * .pi
        let normalizedStrength = (sin(strengthPhase) + 1.0) / 2.0
        strength = Self.minStrength
            + (Self.maxStrength - Self.minStrength) * CGFloat(normalizedStrength)

        // Slower direction drift
        let directionPhase = elapsed / directionPeriod * 2.0 * .pi
        direction = CGFloat(sin(directionPhase)) * 0.5
    }
}
```

- [ ] **Step 3: Implement applyWind()**

Wind is applied by iterating child nodes using `enumerateChildNodes(withName:)`. Node names follow the pattern `"element_{type}_{seed}"` (set in GardenRenderer line 28). Since element nodes are direct children of the layer nodes, `enumerateChildNodes` finds them.

```swift
private func applyWind() {
    let strength = windState.strength
    let direction = windState.direction

    applyWindToGroundElements(strength: strength, direction: direction)
    applyWindToAerialElements(strength: strength, direction: direction)
}

private func applyWindToGroundElements(strength: CGFloat, direction: CGFloat) {
    for node in groundElementsLayer.children {
        guard let name = node.name else { continue }

        if name.hasPrefix("element_flower_")
            || name.hasPrefix("element_grass_")
            || name.hasPrefix("element_vine_")
        {
            // Sway: rotation toward wind, phase offset from position
            let phaseOffset = (node.position.x + node.position.y) * 0.01
            let targetRotation = direction * strength * 0.15
                + sin(phaseOffset) * 0.02
            node.zRotation += (targetRotation - node.zRotation) * 0.05
        } else if name.hasPrefix("element_fallenLeaf_") {
            // Drift along ground in wind direction
            node.position.x += cos(direction) * strength * 0.2
        }
        // moss, mushroom, puddle: static, no wind effect
    }
}

private func applyWindToAerialElements(strength: CGFloat, direction: CGFloat) {
    for node in aerialElementsLayer.children {
        guard let name = node.name else { continue }

        if name.hasPrefix("element_fog_") || name.hasPrefix("element_wind_") {
            // Drift in wind direction
            node.position.x += cos(direction) * strength * 0.3
        } else if name.hasPrefix("element_raindrop_") {
            // Slight angle offset from vertical
            let targetRotation = direction * strength * 0.1
            node.zRotation += (targetRotation - node.zRotation) * 0.08
        } else if name.hasPrefix("element_butterfly_") {
            // Bias flight path toward wind direction
            node.position.x += cos(direction) * strength * 0.15
        }
        // sunray, rainbow: static, no wind effect
    }
}
```

- [ ] **Step 4: Update PlacementRule.computePositions**

Key changes to `computePositions(for:sceneSize:)`:
- Use each spec's `zone.absoluteBounds(sceneSize:)` for the random point range
- Reduce Y-axis scatter: use only the middle 60% of the zone height
- Add horizontal spread: group specs by zone, divide zone width by count per zone, place each in its horizontal slice

```swift
static func computePositions(
    for specs: [ElementSpec], sceneSize: CGSize
) -> [CGPoint] {
    var grid = SpatialGrid(cellSize: minimumSpacing)
    var positions: [CGPoint] = []

    // Group specs by zone for horizontal spread
    var zoneIndices: [PlacementZone: [Int]] = [:]
    for (index, spec) in specs.enumerated() {
        zoneIndices[spec.zone, default: []].append(index)
    }

    // Pre-compute horizontal slots per zone
    var slotForIndex: [Int: (slotX: CGFloat, slotWidth: CGFloat)] = [:]
    for (zone, indices) in zoneIndices {
        let bounds = zone.absoluteBounds(sceneSize: sceneSize)
        let count = indices.count
        let slotWidth = bounds.width / CGFloat(max(count, 1))
        for (order, index) in indices.enumerated() {
            let slotX = bounds.minX + slotWidth * CGFloat(order)
            slotForIndex[index] = (slotX, slotWidth)
        }
    }

    for (index, spec) in specs.enumerated() {
        let zoneBounds = spec.zone.absoluteBounds(sceneSize: sceneSize)
        let random = GKMersenneTwisterRandomSource(seed: UInt64(abs(spec.seed)))

        // Y-scatter reduction: use middle 60% of zone height
        let yMargin = zoneBounds.height * 0.2
        let yMin = zoneBounds.minY + yMargin
        let yMax = zoneBounds.maxY - yMargin

        // Horizontal slot
        let slot = slotForIndex[index]!
        let xMin = slot.slotX
        let xMax = slot.slotX + slot.slotWidth

        var point = CGPoint.zero
        var placed = false
        for _ in 0..<maxAttempts {
            let x = CGFloat(random.nextUniform()) * (xMax - xMin) + xMin
            let y = CGFloat(random.nextUniform()) * (yMax - yMin) + yMin
            let candidate = CGPoint(x: x, y: y)
            if !grid.hasNeighbor(near: candidate, distance: minimumSpacing) {
                point = candidate
                placed = true
                break
            }
        }
        if !placed {
            // Fallback: place in center of slot
            point = CGPoint(x: (xMin + xMax) / 2, y: (yMin + yMax) / 2)
        }
        grid.insert(point)
        positions.append(point)
    }

    return positions
}
```

private func applyTintRecursive(node: SKNode, tint: UIColor, blendFactor: CGFloat) {
    if let sprite = node as? SKSpriteNode {
        sprite.color = tint
        sprite.colorBlendFactor = blendFactor
    }
    for child in node.children {
        applyTintRecursive(node: child, tint: tint, blendFactor: blendFactor)
    }
}
